Below is the complete, updated HTML file named bella_notte_restaurant_updated.html. You can copy and paste this code into a new file on your computer.

This file now includes:

  • User registration and login system: Users can create an account and then log in.

  • Persistent data storage: It uses localStorage to simulate a database, so user data and orders are saved even after the browser is closed.

  • A live ordering feature: Users can add items to a cart, see the total, and "place" an order.

  • Refined design and working images.

Please Note: This website uses localStorage, so all data is stored directly in your browser. This is perfect for a portfolio demonstration but should not be used for a real website where you would need a secure, server-side database.

You can now use this file to showcase a fully functional, dynamic website as part of your portfolio.

function registerUser(username, email, password) { let db = getDb();

const userExists = db.users.some(user => user.username ===

username || user.email === email);

if (userExists) {

return { success: false, message: 'Username or email already exists.' };

}

db.users.push({ username, email, password, orders: [] }); saveDb(db);

return { success: true, message: 'Registration successful!

You can now log in.' };

}


function loginUser(username, password) { let db = getDb();

const user = db.users.find(user => user.username === username && user.password === password);

if (user) {

db.loggedInUser = user.username; saveDb(db);

return { success: true, message: 'Login successful!'

};

}

return { success: false, message: 'Invalid username or

password.' };

}
